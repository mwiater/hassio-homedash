def self.assembleLogMessage(text)
  timestamp = Log.configuration.timestamp ? Time.now.to_s.pale + " " : ""
  prepender = !Log.configuration.prepender.empty? ? Log.configuration.prepender.white + " " : ""

  timestamp << prepender
end
