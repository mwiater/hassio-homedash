configure do
  disable :method_override
  enable :use_code

  set :erb, escape_html: true
  set :protection, except: [:json_csrf]

  # Base views for App
  set :views, ['app/views/pages', 'app/views/layouts']

  set :public_folder, 'app/public'
  set :root, App.root

  puts "Sinatra configuration Complete!".green
